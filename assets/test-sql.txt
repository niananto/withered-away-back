DECLARE
    CURR_VALUE VARCHAR2(50);
BEGIN
    SELECT VACCINE INTO CURR_VALUE FROM HEALTH_RECORD WHERE PEOPLE_ID=:2;
    IF CURR_VALUE IS NULL THEN
        UPDATE HEALTH_RECORD SET VACCINE=:1 WHERE PEOPLE_ID=:2;
    ELSE
        UPDATE HEALTH_RECORD SET VACCINE=(VACCINE || ' , ' || :1) WHERE PEOPLE_ID=:2;
    END IF;
END;
